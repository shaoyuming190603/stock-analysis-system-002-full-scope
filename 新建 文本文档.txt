from PySide6.QtWidgets import QApplication, QMainWindow, QWidget, QRadioButton, QTabWidget, QVBoxLayout
import sys

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        # 假设你已经通过UI文件加载了控件
        # self.ui = Ui_MainWindow()
        # self.ui.setupUi(self)
        # 这里用代码演示
        central = QWidget()
        layout = QVBoxLayout(central)
        self.radioButton1 = QRadioButton("Tab 1")
        self.radioButton2 = QRadioButton("Tab 2")
        self.tabWidget_mainWindow = QTabWidget()
        self.tabWidget_mainWindow.addTab(QWidget(), "Tab 1")
        self.tabWidget_mainWindow.addTab(QWidget(), "Tab 2")
        layout.addWidget(self.radioButton1)
        layout.addWidget(self.radioButton2)
        layout.addWidget(self.tabWidget_mainWindow)
        self.setCentralWidget(central)

        # 默认选中第一个
        self.radioButton1.setChecked(True)

        # 绑定信号
        self.radioButton1.toggled.connect(self.on_radio1_toggled)
        self.radioButton2.toggled.connect(self.on_radio2_toggled)
        self.tabWidget_mainWindow.currentChanged.connect(self.on_tab_changed)

    def on_radio1_toggled(self, checked):
        if checked:
            self.tabWidget_mainWindow.setCurrentIndex(0)

    def on_radio2_toggled(self, checked):
        if checked:
            self.tabWidget_mainWindow.setCurrentIndex(1)

    def on_tab_changed(self, index):
        if index == 0:
            self.radioButton1.setChecked(True)
        elif index == 1:
            self.radioButton2.setChecked(True)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = MainWindow()
    window.show()
    sys.exit(app.exec())
